370  #%% Conversão da estrutura de séries para aprendizado supervisionado
371  # convert series to supervised learning
372  def series_to_supervised(self, data, n_in=1, n_out=1, dropnan=True):
373  """
374  Função que transforma os dados de séries temporais para em sequências
375  que possam ser entendidas pela rede neural.
376 
377  Parameters
378  ----------
379  data : pd.DataFrame,
380  Séries Temporais no formato de um DataFrame, onde a primeira coluna
381  representa a variável eendógena e as seguintes as variáveis exógenas.
382  n_in : int, optional
383  Número de dados de entrada (t-n_in) para previsão do dado em t. The default is 1.
384  n_out : int, optional
385  Número de dados de saída (em t+n_out-1). The default is 1.
386  dropnan : bool, optional
387  Variável booleana. True para excluir NaN e False caso contrário. The default is True.
388 
389  Returns
390  -------
391  agg : pd.DataFrame,
392  Séries na forma de sequências para interpretação da rede neural.
393 
394  """
395  n_vars = 1 if type(data) is list else data.shape[1]
396  df = DataFrame(data)
397  cols, names = list(), list()
398  # input sequence (t-n, ... t-1)
399  for i in range(n_in, 0, -1):
400  cols.append(df.shift(i))
401  names += [('var%d(t-%d)' % (j+1, i)) for j in range(n_vars)]
402  # forecast sequence (t, t+1, ... t+n)
403  for i in range(0, n_out):
404  cols.append(df.shift(-i))
405  if i == 0:
406  names += [('var%d(t)' % (j+1)) for j in range(n_vars)]
407  else:
408  names += [('var%d(t+%d)' % (j+1, i)) for j in range(n_vars)]
409  # put it all together
410  agg = concat(cols, axis=1)
411  agg.columns = names
412  # drop rows with NaN values
413  if dropnan:
414  agg.dropna(inplace=True)
415  return agg
